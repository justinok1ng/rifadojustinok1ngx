document.addEventListener('DOMContentLoaded', function () {

    const finalDateISO = document.querySelector('.double-chance-date-final').getAttribute('data-final').trim();

    const initialDate = new Date(initialDateISO);
    const finalDate = new Date(finalDateISO);

    if (isNaN(initialDate.getTime()) || isNaN(finalDate.getTime())) {
        console.error("Datas inv√°lidas.");
        return;
    }

    const formatDate = (date) => {
        const day = String(date.getDate()).padStart(2, '0');
        const month = String(date.getMonth() + 1).padStart(2, '0');
        const year = date.getFullYear();
        const hours = String(date.getHours()).padStart(2, '0');
        const minutes = String(date.getMinutes()).padStart(2, '0');
        return `${day}/${month}/${year} ${hours}:${minutes}`;
    };

    document.querySelector('.double-chance-date-initial').innerText = formatDate(initialDate);
    document.querySelector('.double-chance-date-final').innerText = formatDate(finalDate);

    const totalTime = finalDate - initialDate;

    const timerElement = document.createElement('p');
    timerElement.classList.add('timer');
    document.querySelector('.progress-bar-double-chance').appendChild(timerElement);

    function updateTimer() {
        const now = new Date();
        const timeRemaining = finalDate - now; 

        if (timeRemaining <= 0) {
            document.querySelector('.progress-bar-inner').style.width = '100%';
            timerElement.innerText = 'PROMO√á√ÉO FINALIZADA!';
            return;
        }

        const secondsRemaining = Math.floor(timeRemaining / 1000);
        const hours = Math.floor(secondsRemaining / 3600); 
        const minutes = Math.floor((secondsRemaining % 3600) / 60); 
        const seconds = secondsRemaining % 60; 
        const formattedTime = `ENCERRA EM ${String(hours).padStart(2, '0')}:${String(minutes).padStart(2, '0')}:${String(seconds).padStart(2, '0')}`;

        const progressWidth = (1 - timeRemaining / totalTime) * 100;
        document.querySelector('.progress-bar-inner').style.width = `${progressWidth}%`;

        timerElement.innerText = formattedTime;
    }

    setInterval(updateTimer, 1000);
});

  // ====== TIMER PIX ======
  let pixTimerInterval = null;
  function startPixTimer(seconds){
    if(!pixTimerEl) return;
    if(pixTimerInterval) clearInterval(pixTimerInterval);

    let remain = Math.max(0, parseInt(seconds || 0, 10));

    function tick(){
      const mm = String(Math.floor(remain / 60)).padStart(2,'0');
      const ss = String(remain % 60).padStart(2,'0');
      pixTimerEl.textContent = `${mm}:${ss}`;
      remain--;
      if(remain < 0){
        clearInterval(pixTimerInterval);
        pixTimerInterval = null;
      }
    }

    tick();
    pixTimerInterval = setInterval(tick, 1000);
  }

  // ====== COPY PIX ======
  async function copyPixText(){
    const text = (pixCopyPaste?.textContent || '').trim();
    if(!text || text === '‚Äî') return;

    try{
      await navigator.clipboard.writeText(text);
      if(pixCopyBtn){
        const old = pixCopyBtn.textContent;
        pixCopyBtn.textContent = 'COPIADO ‚úì';
        setTimeout(()=> pixCopyBtn.textContent = old, 1200);
      }
    }catch(e){
      // fallback
      const ta = document.createElement('textarea');
      ta.value = text;
      document.body.appendChild(ta);
      ta.select();
      document.execCommand('copy');
      document.body.removeChild(ta);
    }
  }

  if(pixCopyBtn){
    pixCopyBtn.addEventListener('click', function(e){
      e.preventDefault();
      copyPixText();
    });
  }

      pixToggleBtn.textContent = isOpen ? 'Mostrar QR Code' : 'Ocultar QR Code';
    });

          phone,
          name,
          qty,

        });

        if(pixPhone) pixPhone.textContent = phone || '‚Äî';
        if(pixQty) pixQty.textContent = String(qty || '‚Äî');
        if(pixTotal) pixTotal.textContent = moneyBR(total);
        if(pixProductCard) pixProductCard.textContent = product;

        startPixTimer(data.expiresInSeconds || 545);

        showPix();
      }catch(err){
        console.error(err);
        hideAllFullScreens();
        alert('Erro ao gerar PIX. Tente novamente.');
      }
    });
  }
